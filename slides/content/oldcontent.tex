\begin{frame}[t]{Strategies - Best Interface First}
	\begin{center}
	\includegraphics[width=1\linewidth]{images/forwarding_bif.pdf}%	\\[3mm]
	\end{center}
%	Examples: CCNx Default, NDN BestRoute,\\ Yi et al. -- \emph{A case for stateful forwarding plane (2013)}, \emph{INFORM} (2013), \emph{PAF} (2013)
\end{frame}

\note[enumerate]{
	\item Send on best path first. On packet loss: try other paths.
	\item CCNx Default: Switching path on packet loss,  Interface probing to find better path!
	\item NDN BestRoute: Use interface with smallest routing costs.
	\item Yi et al: Coloring scheme, Interest NACK.
}

\begin{frame}[t]{Strategies - Packet Striping}
	%\vspace{-4mm}
	\begin{center}
	\includegraphics[width=1\linewidth]{images/forwarding_packet_striping.pdf}%\\[3mm]
	\end{center}
%	Examples: CCNx Loadsharing, Carofiglio et al. -- \\ \emph{Optimal Multipath Congestion Control and Request Forwarding in ICN} (2013), \emph{OMP-IF} (2014), \emph{SAF} (2015)
\end{frame}

\note[enumerate]{
	\item Split packets to improve overall throughput
	\item CCNx Loadsharing: Minimize outstanding Interests, Faster paths get more packets
	\item Carofiglio et al. (Bell Labs, Orange Labs): Fair loadbalancing with competing traffic
}

\begin{frame}[t]{Strategies - Parallel Transmission}
	\begin{center}
	\includegraphics[width=1\linewidth]{images/forwarding_parallel.pdf}%\\[3mm]
	\end{center}
%	Examples: CCNx Parallel,\\ Rossini et al. -- Multipath parallel (2013)
\end{frame}

\note[enumerate]{
	\item Sending Interest packets redundantly, accepting the first returning data packets and ignoring later arriving ones.
	\item CCNx Parallel: Only simplistic flooding = Send every packet to every interface
	\item Flooding is often unnecessarily wasteful $\rightarrow$ Selective redundancy.
}
